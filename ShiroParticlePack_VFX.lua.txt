local function posGrid(idx)
    local col = (idx-1) % CFG.COLUNAS
    local row = math.floor((idx-1) / CFG.COLUNAS)
    return CFG.INICIO + Vector3.new(col * CFG.ESPACO, 0, row * CFG.ESPACO)
end

local function applyPE(pe, props)
    local campos = {
        "Texture","Color","LightEmission","LightInfluence","Size","Transparency",
        "Rate","Lifetime","Speed","SpreadAngle","Rotation","RotSpeed",
        "EmissionDirection","LockedToPart","ZOffset","Acceleration",
        "FlipbookMode","FlipbookLayout","FlipbookFramerate","FlipbookStartRandom",
        "ShapeStyle","ShapeInOut","ShapePartial","TimeScale","VelocityInheritance",
    }
    for _, campo in ipairs(campos) do
        if props[campo] ~= nil then
            pcall(function() pe[campo] = props[campo] end)
        end
    end
end

local function criarVFX(vfxDef, idx)
    local pos = posGrid(idx)

    local pasta = Instance.new("Folder")
    pasta.Name = vfxDef.nome

    local parts = {}
    if vfxDef.parts and #vfxDef.parts > 0 then
        for i, pd in ipairs(vfxDef.parts) do
            local p = Instance.new("Part")
            p.Name = pd.nome or (vfxDef.nome .. "_Part" .. i)
            p.Size = pd.tamanho or Vector3.new(0.4, 0.4, 0.4)
            p.CFrame = CFrame.new(pos)
            p.Anchored = pd.anchored ~= false
            p.CanCollide = false
            p.CanQuery = false
            p.CastShadow = false
            p.Material = pd.material or Enum.Material.SmoothPlastic
            pcall(function() p.Color = pd.cor or Color3.fromRGB(255,255,255) end)
            pcall(function() p.Transparency = pd.transparencia or 0 end)
            pcall(function()
                if pd.forma == Enum.PartType.Ball then
                    p.Shape = Enum.PartType.Ball
                elseif pd.forma == Enum.PartType.Cylinder then
                    p.Shape = Enum.PartType.Cylinder
                end
            end)
            p.Parent = pasta
            table.insert(parts, p)
        end
    else
        local p = Instance.new("Part")
        p.Name = vfxDef.nome .. "_Anchor"
        p.Size = Vector3.new(0.3, 0.3, 0.3)
        p.CFrame = CFrame.new(pos)
        p.Anchored = true
        p.CanCollide = false
        p.Transparency = 1
        p.Parent = pasta
        table.insert(parts, p)
    end

    local partePrincipal = parts[1]

    if vfxDef.particulas then
        for i, pDef in ipairs(vfxDef.particulas) do
            local pe = Instance.new("ParticleEmitter")
            pe.Name = pDef.nome or ("Particle_" .. i)
            applyPE(pe, pDef)
            pe.Parent = partePrincipal
        end
    end

    -- Label flutuante
    local bb = Instance.new("BillboardGui")
    bb.Name = "Label"
    bb.Size = UDim2.new(0, 90, 0, 20)
    bb.StudsOffset = Vector3.new(0, 1.5, 0)
    bb.AlwaysOnTop = true
    bb.Adornee = partePrincipal
    bb.Parent = partePrincipal

    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(1,0,1,0)
    lbl.BackgroundColor3 = Color3.fromRGB(0,0,0)
    lbl.BackgroundTransparency = 0.45
    lbl.Text = vfxDef.nome:gsub("_", " ")
    lbl.TextColor3 = Color3.fromRGB(255,255,255)
    lbl.TextScaled = true
    lbl.Font = Enum.Font.GothamBold
    lbl.Parent = bb

    local lblCorner = Instance.new("UICorner")
    lblCorner.CornerRadius = UDim.new(0,4)
    lblCorner.Parent = lbl

    return pasta
end

local function garantirPasta()
    if PASTA_RAIZ and PASTA_RAIZ.Parent then return PASTA_RAIZ end
    local ex = workspace:FindFirstChild(CFG.PASTA_NOME)
    if ex then ex:Destroy() end
    PASTA_RAIZ = Instance.new("Folder")
    PASTA_RAIZ.Name = CFG.PASTA_NOME
    PASTA_RAIZ.Parent = workspace
    return PASTA_RAIZ
end

local function setStatus(txt, cor)
    STATUS.Text = txt
    STATUS.TextColor3 = cor or Color3.fromRGB(160,160,255)
end

local function setProgress(n, total)
    local pct = n / total
    PBAR:TweenSize(UDim2.new(pct, 0, 1, 0), "Out", "Quad", 0.3, true)
end

local function updateItem(nome, estado)
    local f = itemFrames[nome]
    if not f then return end
    local dot = f:FindFirstChild("Dot")
    if not dot then return end
    if estado == "ok" then
        dot.BackgroundColor3 = Color3.fromRGB(0,220,80)
    elseif estado == "loading" then
        dot.BackgroundColor3 = Color3.fromRGB(255,200,0)
    else
        dot.BackgroundColor3 = Color3.fromRGB(60,60,60)
    end
end

B_START.MouseButton1Click:Connect(function()
    if carregando then return end
    carregando = true
    carregados = 0
    B_START.BackgroundColor3 = Color3.fromRGB(40,40,80)

    task.spawn(function()
        local pasta = garantirPasta()
        local total = #VFX_PACK

        setStatus("â³ Carregando " .. total .. " VFX â€” 0.6s cada...", Color3.fromRGB(255,200,80))
        task.wait(0.3)

        for i, vfxDef in ipairs(VFX_PACK) do
            if not carregando then break end

            if instancias[vfxDef.nome] then
                carregados += 1
                setProgress(carregados, total)
                continue
            end

            updateItem(vfxDef.nome, "loading")
            setStatus("â³ ["..i.."/"..total.."]  "..vfxDef.nome:gsub("_"," "), Color3.fromRGB(255,220,80))

            local ok, result = pcall(criarVFX, vfxDef, i)
            if ok and result then
                result.Parent = pasta
                instancias[vfxDef.nome] = result
                updateItem(vfxDef.nome, "ok")
            else
                setStatus("âš  Erro: "..tostring(result), Color3.fromRGB(255,80,80))
                task.wait(0.4)
            end

            carregados += 1
            setProgress(carregados, total)
            task.wait(CFG.DELAY) -- â† DELAY GRADUAL: evita crash mobile
        end

        carregando = false
        B_START.BackgroundColor3 = Color3.fromRGB(80,30,180)
        setStatus("âœ… "..carregados.."/"..total.." VFX na pasta '"..CFG.PASTA_NOME.."'!", Color3.fromRGB(80,255,150))
    end)
end)
